<script type="module">
  import Typebot from 'https://cdn.jsdelivr.net/npm/@typebot.io/js@0.3/dist/web.js'

  Typebot.initBubble({
    typebot: "DEIN-BOT-SLUG",
    apiHost: "https://bot.eppcom.de",
    theme: {
      button: { backgroundColor: "#1D1D1D" },
      chatWindow: { backgroundColor: "#F8F8F8" }
    },
    previewMessage: {
      message: "Wie kann ich Ihnen helfen?",
      autoShowDelay: 5000
    }
  })
</script>
